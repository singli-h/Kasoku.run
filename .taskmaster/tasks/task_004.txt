# Task 4: Implement Proper Authentication Patterns

## Description
Audit and fix any direct clerk_id usage, implement consistent authentication error handling, and add proper loading states for authentication flows.

## Details
1. Create a utility function to map clerk_id to user.id consistently.
2. Audit all components and server actions for direct clerk_id usage and replace with user.id.
3. Implement a custom AuthProvider component that handles authentication state and provides a consistent interface.
4. Add loading states for authentication flows using react-loading-skeleton v3.3.1.
5. Implement consistent error handling for authentication errors.

Example implementation:
```typescript
// lib/auth-utils.ts
import { createClientComponentClient } from '@supabase/auth-helpers-nextjs';

const supabase = createClientComponentClient();

export async function getAuthenticatedUserId(): Promise<string | null> {
  const { data: { user } } = await supabase.auth.getUser();
  return user?.id || null;
}

// components/AuthProvider.tsx
import { createContext, useContext, useState, useEffect } from 'react';
import { useUser } from '@clerk/nextjs';
import { getAuthenticatedUserId } from '../lib/auth-utils';
import Skeleton from 'react-loading-skeleton';

const AuthContext = createContext<{ userId: string | null; loading: boolean }>({ userId: null, loading: true });

export const AuthProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const { isLoaded, isSignedIn } = useUser();
  const [userId, setUserId] = useState<string | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    if (isLoaded && isSignedIn) {
      getAuthenticatedUserId().then(id => {
        setUserId(id);
        setLoading(false);
      }).catch(error => {
        console.error('Failed to get authenticated user ID:', error);
        setLoading(false);
      });
    } else if (isLoaded) {
      setLoading(false);
    }
  }, [isLoaded, isSignedIn]);

  if (loading) return <Skeleton count={5} />;

  return <AuthContext.Provider value={{ userId, loading }}>{children}</AuthContext.Provider>;
};

export const useAuth = () => useContext(AuthContext);
```

Usage in components:
```typescript
import { useAuth } from '../components/AuthProvider';

const MyComponent: React.FC = () => {
  const { userId, loading } = useAuth();

  if (loading) return <Skeleton count={3} />;
  if (!userId) return <div>Please sign in to access this feature.</div>;

  // Rest of the component logic
};
```
