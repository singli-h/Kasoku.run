# Task 7: Implement Performance Tracking for Athletes

## Description
Develop a reliable performance tracking system for athletes within the Sessions page.

## Details
1. Design and implement a data structure for storing athlete performance metrics.
2. Create API endpoints for recording and retrieving performance data.
3. Implement real-time updates using Supabase realtime subscriptions.
4. Develop visualizations for performance data using a charting library like Chart.js or D3.js.
5. Implement performance comparison features between sessions and athletes.

Example implementation:
```typescript
// types/performance.ts
interface PerformanceMetric {
  id: string;
  athleteId: string;
  sessionId: string;
  metric: string;
  value: number;
  timestamp: Date;
}

// lib/performance-api.ts
import { createClientComponentClient } from '@supabase/auth-helpers-nextjs';

const supabase = createClientComponentClient();

export async function recordPerformance(metric: Omit<PerformanceMetric, 'id'>): Promise<PerformanceMetric> {
  const { data, error } = await supabase
    .from('performance_metrics')
    .insert(metric)
    .select()
    .single();

  if (error) throw new ApiError(error.message, error.code);
  return data;
}

export async function getAthletePerformance(athleteId: string, sessionId: string): Promise<PerformanceMetric[]> {
  const { data, error } = await supabase
    .from('performance_metrics')
    .select('*')
    .eq('athleteId', athleteId)
    .eq('sessionId', sessionId);

  if (error) throw new ApiError(error.message, error.code);
  return data;
}

// components/PerformanceChart.tsx
import { Line } from 'react-chartjs-2';
import { useSupabaseQuery } from '../hooks/useSupabaseQuery';

const PerformanceChart: React.FC<{ athleteId: string, sessionId: string }> = ({ athleteId, sessionId }) => {
  const { data, isLoading, error } = useSupabaseQuery(
    ['performance', athleteId, sessionId],
    () => getAthletePerformance(athleteId, sessionId)
  );

  if (isLoading) return <Skeleton height={200} />;
  if (error) return <div>{handleError(error)}</div>;

  const chartData = {
    labels: data.map(d => d.timestamp),
    datasets: [{
      label: 'Performance',
      data: data.map(d => d.value),
      borderColor: 'rgb(75, 192, 192)',
      tension: 0.1
    }]
  };

  return <Line data={chartData} />;
};

// Usage in SessionDetails component
const SessionDetails: React.FC<{ sessionId: string }> = ({ sessionId }) => {
  const { athletes } = useSessionAthletes(sessionId);

  return (
    <div>
      <h2>Athlete Performance</h2>
      {athletes.map(athlete => (
        <PerformanceChart key={athlete.id} athleteId={athlete.id} sessionId={sessionId} />
      ))}
    </div>
  );
};
```
